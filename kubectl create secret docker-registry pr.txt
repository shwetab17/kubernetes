kubectl create secret docker-registry  private-reg-cred \
 --docker-server=myprivateregistry.com:5000 --docker-username=dock_user \
  --docker-password=dock_password --docker-email=dock_user@myprivateregistry.com





pipeline {
    agent any

    stages {
        stage('Build') {
            steps {
                // Get some code from a GitHub repository
                
                sh "rm -rf simple-maven-project-with-tests; git clone 'https://github.com/jglick/simple-maven-project-with-tests.git'"
                sh "mvn -version"
                sh "echo current dir is: `pwd` "
                sh "cd simple-maven-project-with-tests; mvn package -Dskiptests"
            }

            post {
                // If Maven was able to run the tests, even if some of the test
                // failed, record the test results and archive the jar file.
                success {
                    junit '**/target/surefire-reports/TEST-*.xml'
                    archiveArtifacts 'target/*.jar'
                    copyArtifacts(target: '/mnt/artefact');
                }
            }
        }
    }
}
